cmake_minimum_required(VERSION 3.24)

project(glad VERSION 0.1.36 DESCRIPTION "OpenGL loader generated by glad 0.1.36.")

include(GNUInstallDirs)

option(GLAD_STATIC "build glad as a static library" ON)

if(GLAD_STATIC)
	add_library(glad STATIC
			src/glad.c
	)
else ()
	add_library(glad SHARED
			src/glad.c
	)
endif ()


set_target_properties(glad PROPERTIES VERSION ${PROJECT_VERSION})
set_target_properties(glad PROPERTIES SOVERSION 0)

target_include_directories(glad PUBLIC include)

set_property(TARGET glad PROPERTY POSITION_INDEPENDENT_CODE ON)
